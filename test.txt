CREATE TABLE `t_mission` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `name` varchar(31) NOT NULL DEFAULT '' COMMENT '任务名，如每日登录奖励',
  `type` tinyint(3) unsigned NOT NULL DEFAULT '1' COMMENT '类型 1-每日登录奖励，2-初始资本，3-分享获得',
  `fixed` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '固定奖励多少铜币',
  `min` bigint(20) unsigned NOT NULL DEFAULT '0' COMMENT '奖励最少铜币，连续型任务',
  `max` bigint(20) unsigned NOT NULL DEFAULT '0' COMMENT '奖励最多铜币，连续型任务',
  `incr` int(10) unsigned NOT NULL DEFAULT '0' COMMENT '连续登录增量，连续型任务',
  `state` tinyint(3) unsigned NOT NULL DEFAULT '0' COMMENT '状态: 0-正常，未完成；1-已过期；2-已下线',
  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COMMENT='任务表'
